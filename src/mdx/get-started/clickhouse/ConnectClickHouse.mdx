```ts
import { waddler } from "waddler/clickhouse";
import { createClient } from '@clickhouse/client';
  
const client = createClient({
  url: '...',
  // url should have the following format: http[s]://<host>:<port>
  user: 'user',
  password: 'password',
  database: 'database',
});
const sql = waddler({ client });
```
