**Streaming**

```typescript copy filename="src/index.ts"
import 'dotenv/config';
import { waddler } from 'waddler/duckdb-neo';

async function main() {
	const sql = waddler({ url: process.env.DATABASE_URL!, max: 10, accessMode: 'read_write' });

	const stream = sql`select * from users;`.stream();

	console.log('Streaming users one at a time from the database.')
  for await (const user of stream) {
      console.log(user)
      /*
      const user: {
        id: number;
        name: string;
        age: number;
        email: string;
      }
      */		
	}
}

main();
```
